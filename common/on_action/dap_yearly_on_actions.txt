# Called from code once a year for "playable" (count+) characters.
# Root is the character
yearly_playable_pulse = {
    on_actions = { dap_yearly_interest_increase }
}

dap_yearly_interest_increase = {
    effect = {
        if = {
            limit = { 
                has_character_flag = dap_borrow_from_jewish_merchants 
            }
            change_variable = {
                name = dap_amount_to_loan_jew
                multiply = 1.3
            }
        }
        else = {
            change_variable = {
                name = dap_amount_to_loan_jew
                add = 60
            }
        }
    }
}
